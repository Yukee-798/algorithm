function reverse(x: number): number {
    let isNegative = (x + '')[0] === '-', 
        numStr = x + '', 
        ans: number;
    if (isNegative) {
        numStr = numStr.substring(1);
        ans = +('-' + [...numStr].reverse().join(''));
    } else {
        ans = +[...numStr].reverse().join('');
    }
    return ans <= (((1 << 30) - 1) * 2 + 1) && ans >= (-(1 << 30) * 2) ? ans : 0;
};
